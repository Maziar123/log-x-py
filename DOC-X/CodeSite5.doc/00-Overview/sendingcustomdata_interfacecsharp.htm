Sending Custom Data: ICodeSiteCustomData (C#)

    

Note: *Not Available in CodeSite Express

The following example code illustrates how to implement the ICodeSiteCustomData interface in C#:

class BaseballPlayer: ICodeSiteCustomData

{

  public string LastName = "";

  public string FirstName = "";

  public string Position = "";

  public Decimal Salary;

  public int Games;

  public int AtBats;

  public int Hits;

  public Single ERA;

  public bool AllStar;

 

  public StringBuilder GetFormattedData()

  {

    StringBuilder sb = new StringBuilder();

   

    // The following code formats the player data in a text format

 

    sb.AppendFormat( "{0} {1}\r\n", FirstName, LastName );

    if ( AllStar ) 

      sb.AppendFormat( "All-Star {0}\r\n", Position );

    else

      sb.AppendFormat( "{0}\r\n", Position );

    sb.Append( "\r\n" );

    sb.AppendFormat( "Salary: {0:C}\r\n", Salary );

    sb.Append( "\r\n" );

    sb.AppendFormat( "Games Played: {0}\r\n", Games );

 

    if ( Position == "Pitcher" )

      sb.AppendFormat( "ERA: {0:N}\r\n", ERA );

    else

      sb.AppendFormat( "Batting Avg: {0:0.000}\r\n", (float) Hits / AtBats );

 

    // The following code can be used to format the player data in 

    // a name-value property grid

    /*

    sb.AppendFormat( "LastName={0}\r\n", LastName );

    sb.AppendFormat( "FirstName={0}\r\n", FirstName );

    sb.AppendFormat( "Position={0}\r\n", Position );

    sb.AppendFormat( "Salary={0:C}\r\n", Salary );

    sb.AppendFormat( "Games={0}\r\n", Games );

    sb.AppendFormat( "AtBats={0}\r\n", AtBats );

    sb.AppendFormat( "Hits={0}\r\n", Hits );

    sb.AppendFormat( "ERA={0:0.00}\r\n", ERA );

    if ( AllStar ) 

      sb.AppendFormat( "AllStar=True\r\n" );

    else

      sb.AppendFormat( "AllStar=False\r\n" );

    */

   

    return sb;

  }

 

 

  public CodeSiteInspectorType InspectorType()

  {

    return CodeSiteInspectorType.StockProperties;

  }

 

 

  public int MsgType()

  { 

    // Simply return 0 when NOT specifying a Custom inspector type

    return 0;

  }

 

 

  public string TypeName()

  {

    return "BaseballPlayer";

  }

}

 

The above class could then be used as follows:

{

  BaseballPlayer player = new BaseballPlayer();

 

  player.LastName = "Randy";

  player.FirstName = "Johnson";

  player.Position = "Pitcher";

  player.Salary = 4700000.00M;

  player.Games = 29;

  player.AtBats = 50;

  player.Hits = 5;

  player.ERA = 1.49F;

  player.AllStar = true;

 

  // The Send method will determine if the object supports

  // the ICodeSiteCustomData interface and call its methods as needed

  CodeSite.Send( "Current Player", player );

}
